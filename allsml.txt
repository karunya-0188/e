1. Leaky Bucket
#include <stdio.h>

int main() {
    int bs, rate, n, in, store = 0;
    scanf("%d%d%d", &bs, &rate, &n);

    while (n--) {
        scanf("%d", &in);
        if (in + store > bs) store = bs;
        else store += in;

        store -= rate;
        if (store < 0) store = 0;

        printf("%d\n", store);
    }
    return 0;
}

5. Distance Vector Routing (DVR)
#include <stdio.h>

#define N 4
#define INF 999

int main() {
    int c[N][N]={{0,1,3,7},{1,0,1,INF},{3,1,0,2},{7,INF,2,0}}, d[N][N],i,j,k;

    for(i=0;i<N;i++)
        for(j=0;j<N;j++)
            d[i][j] = c[i][j];

    for(k=0;k<N;k++)
        for(i=0;i<N;i++)
            for(j=0;j<N;j++)
                if (d[i][k] + d[k][j] < d[i][j])
                    d[i][j] = d[i][k] + d[k][j];

    for(i=0;i<N;i++){
        for(j=0;j<N;j++)
            printf("%d ", d[i][j]);
        puts("");
    }
}

7. Shortest Path Routing (Dijkstra) 
#include <stdio.h>

#define N 5
#define INF 999

int main() {
    int g[N][N]={{0,10,0,30,100},{10,0,50,0,0},{0,50,0,20,10},{30,0,20,0,60},{100,0,10,60,0}};
    int d[N], v[N]={0}, i, j, u, min;

    for(i=0;i<N;i++) d[i]=INF;
    d[0]=0;

    for(i=0;i<N;i++){
        min = INF;
        for(j=0;j<N;j++)
            if(!v[j] && d[j] < min)
                min = d[j], u = j;

        v[u] = 1;

        for(j=0;j<N;j++)
            if(g[u][j] && d[u] + g[u][j] < d[j])
                d[j] = d[u] + g[u][j];
    }

    for(i=0;i<N;i++)
        printf("%d ", d[i]);
}
